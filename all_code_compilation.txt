# 01_download_pubmedqa.py
from datasets import load_dataset
import pandas as pd
from pathlib import Path

out_dir = Path("data_samples"); out_dir.mkdir(parents=True, exist_ok=True)

ds = load_dataset("pubmed_qa", "pqa_labeled")  # small, labeled variant
train = ds["train"]

# keep a tiny sample for quick baselines
df = pd.DataFrame({
    "question": [x["question"] for x in train[:500]],
    "context": [x["context"] for x in train[:500]],
    "long_answer": [x["long_answer"] for x in train[:500]],
    "final_decision": [x["final_decision"] for x in train[:500]],  # yes/maybe/no
})
df.to_csv(out_dir / "pubmedqa_500.csv", index=False)
print("Saved:", out_dir / "pubmedqa_500.csv", "rows:", len(df))

import pandas as pd
import matplotlib.pyplot as plt

# Load dataset
df = pd.read_csv("data_samples/pubmedqa_500.csv")

# Overview
print("Rows:", len(df))
print("\nColumns:", df.columns.tolist())
print("\nSample Question:\n", df['question'].iloc[0])
print("\nSample Context (truncated):\n", df['context'].iloc[0][:400])

# Label distribution
if 'final_decision' in df.columns:
    label_counts = df['final_decision'].value_counts()
    print("\nLabel distribution:\n", label_counts)
    
    # Plot
    label_counts.plot(kind='bar', title='Answer Distribution (Yes/No/Maybe)')
    plt.ylabel('Count')
    plt.show()

# Average text lengths
df['q_len'] = df['question'].str.len()
df['c_len'] = df['context'].str.len()
print("\nAverage Question Length:", df['q_len'].mean())
print("Average Context Length:", df['c_len'].mean())
